
 SELECT
 DOCINFO.DI_KEY, 
 DOCINFO.DI_REF,
 DOCINFO.DI_DATE,
 DOCINFO.DI_ACTIVE,
 ARFILE.AR_CODE,
 ARFILE.AR_NAME 
 FROM
 DOCINFO,
 DOCTYPE,
 AROE,
 ARFILE WHERE 
 DOCINFO.DI_DT = DOCTYPE.DT_KEY AND
 DOCTYPE.DT_PROPERTIES = 207 AND
 DOCINFO.DI_KEY = AROE.AROE_DI AND
 AROE.AROE_AR = ARFILE.AR_KEY  AND DI_REF like 'BKSV%' AND YEAR(DI_DATE) = 2022 AND MONTH(DI_DATE) = 03 ORDER BY DOCINFO.DI_KEY
